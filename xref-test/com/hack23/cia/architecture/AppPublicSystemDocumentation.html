36  	public static final class SpringVaadinViewComponentFinderStrategy extends AbstractSpringComponentFinderStrategy {
37  
38  	    private static final String SPRING_VIEW = "Vaadin View";
39  
40  		public SpringVaadinViewComponentFinderStrategy(SupportingTypesStrategy... strategies) {
41  	        super(strategies);
42  	    }
43  
44  	    @Override
45  	    protected Set<Component> doFindComponents() throws Exception {
46  	        return findInterfacesForImplementationClassesWithAnnotation(
47  	        		org.springframework.stereotype.Repository.class,
48  	                SPRING_VIEW
49  	        );
50  	    }
51  
52  	}

54  	public static void main(String[] args) throws Exception {
55  		Workspace workspace = new Workspace("Citizen Intelligence Agency", "Public System Documentation");
56  		Model model = workspace.getModel();
57  		ViewSet viewSet = workspace.getViews();
58  		
59  		Person userPerson = model.addPerson("User", "User of the system");
60  		Person adminPerson = model.addPerson("Admin", "Manager of the system");
61  
62  		SoftwareSystem ciaSystem = model.addSoftwareSystem("Citizen Intelligence Agency System",
63  				"Tracking politicians like bugs!");
64  
65  		SoftwareSystem riksdagenApiSystem = model.addSoftwareSystem(Location.External, "data.riksdagen.se","Public API Swedish Parliament data");
66  		SoftwareSystem worldBankApiSystem = model.addSoftwareSystem(Location.External, "data.wordbank.org","Public API Country indicators");
67  		SoftwareSystem valApiSystem = model.addSoftwareSystem(Location.External, "www.val.se","Public API Swedish Election data");
68  		SoftwareSystem esvApiSystem = model.addSoftwareSystem(Location.External, "www.esv.se","Public Data Swedish public sector spending data");
69  
70  		Container loadBalancerContainer = ciaSystem.addContainer("Loadbalancer", "Loadbalancer", "ALB/ELB/Apache/Nginx/HaProxy");
71  	
72  		Container ciaWebContainer = ciaSystem.addContainer("Web Application", "Web Application", "Jetty/Java");
73  		ComponentFinder componentFinderWeb = new ComponentFinder(ciaWebContainer,
74  				"com.hack23.cia", new SpringServiceComponentFinderStrategy(),new SpringComponentComponentFinderStrategy(), new SpringVaadinViewComponentFinderStrategy());
75  		componentFinderWeb.exclude(".*pagemode.*");
76  		componentFinderWeb.exclude(".*common.*");
77  		componentFinderWeb.exclude(".*action.*");
78  		componentFinderWeb.exclude(".*listener.*");
79  		componentFinderWeb.exclude(".*ui.*");
80  		componentFinderWeb.exclude(".*package.*");				
81  		
82  		componentFinderWeb.findComponents();
83  		
84  		Container relationalDatabase = ciaSystem.addContainer("Database", "Stores information", "Postgresql");
85  		relationalDatabase.addTags("Database");
86  
87  		
88  		adminPerson.uses(ciaSystem, "Manages");
89  		userPerson.uses(ciaSystem, "Uses");
90  		ciaSystem.uses(riksdagenApiSystem, "Loads data");
91  		ciaSystem.uses(worldBankApiSystem, "Loads data");
92  		ciaSystem.uses(valApiSystem, "Loads data");	
93  		ciaSystem.uses(esvApiSystem, "Loads data");			
94  		loadBalancerContainer.uses(ciaWebContainer, "HTTPS/H2");
95  		
96  		ciaWebContainer.uses(relationalDatabase, "JDBC");		
97  		
98  		viewSet.createEnterpriseContextView("Enterprise", "Enterprise").addAllElements();
99  		viewSet.createSystemContextView(ciaSystem, "System context", "System context").addAllElements();
100 		viewSet.createContainerView(ciaSystem, "Container view", "Application Overview").addAllContainers();
101 		viewSet.createComponentView(ciaWebContainer, "Web", "Web").addAllComponents();
102 
103 		Styles styles = viewSet.getConfiguration().getStyles();
104 		styles.addElementStyle(Tags.COMPONENT).background("#1168bd").color("#ffffff");
105 		styles.addElementStyle(Tags.CONTAINER).background("#1168bd").color("#ffffff");
106 		styles.addElementStyle(Tags.SOFTWARE_SYSTEM).background("#1168bd").color("#ffffff");
107 		styles.addElementStyle(Tags.PERSON).background("#519823").color("#ffffff").shape(Shape.Person);
108 		styles.addElementStyle("Database").shape(Shape.Cylinder);
109 
110 		StructurizrClient structurizrClient = new StructurizrClient(API_KEY, API_SECRET);
111 		structurizrClient.putWorkspace(WORKSPACE_ID, workspace);
112 	}
